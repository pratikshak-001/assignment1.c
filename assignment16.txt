#include <stdio.h>
int main()
{
int n, i;
int a[50], even[50], odd[50];
int e = 0, o = 0;
printf("Enter numberof elements: ");
scanf("%d" , &n);
printf("Enter %d integers:\n", n);
for(i = 0; i < n; i++)
scanf("%d" , &a[i]);
//Partition into even and odd
for(i = 0; i < n; i++)
{
if(a[i] % 2 == 0)
even[e++] = a[i];
else
odd[o++] = a[i];
}
//Display even numbers
printf("\nEven numbers: \n");
for(i = 0; i < e; i++)
printf("%d", even[i]);
//Display odd numbers
printf("\nOdd numbers: \n");
for(i = 0; i < o; i++)
printf("%d", odd[i]);
return 0;
}
